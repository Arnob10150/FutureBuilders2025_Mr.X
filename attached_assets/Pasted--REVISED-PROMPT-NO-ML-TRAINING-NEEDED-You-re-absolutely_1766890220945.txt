   ```javascript
   function matchSymptoms(userInput, symptomDatabase) {
     const tokens = tokenize(userInput.toLowerCase());
     const matches = [];
     
     for (const condition of symptomDatabase) {
       let score = 0;
       let matchedSymptoms = [];
       
       for (const symptom of condition.symptoms) {
         if (tokens.some(token => 
           symptom.includes(token) || 
           levenshteinDistance(token, symptom) < 3
         )) {
           score += 1;
           matchedSymptoms.push(symptom);
         }
       }
       
       if (score > 0) {
         matches.push({
           condition: condition.name,
           confidence: (score / condition.symptoms.length) * 100,
           matchedSymptoms,
           severity: condition.severity,
           recommendations: condition.recommendations
         });
       }
     }
     
     return matches.sort((a, b) => b.confidence - a.confidence);
   }
   ```

```javascript
// Intelligent fuzzy matching
function intelligentMatch(userSymptom, databaseSymptom) {
  // Exact match
  if (userSymptom === databaseSymptom) return 1.0;
  
  // Substring match
  if (userSymptom.includes(databaseSymptom) || 
      databaseSymptom.includes(userSymptom)) return 0.8;
  
  // Levenshtein distance (typo tolerance)
  const distance = levenshteinDistance(userSymptom, databaseSymptom);
  if (distance <= 2) return 0.7;
  
  // Synonym matching
  const synonyms = SYMPTOM_SYNONYMS[databaseSymptom] || [];
  if (synonyms.includes(userSymptom)) return 0.9;
  
  return 0;
}
```

### **2. Confidence Calculation**
```javascript
function calculateConfidence(matchedSymptoms, totalSymptoms, severity) {
  const matchRatio = matchedSymptoms.length / totalSymptoms.length;
  const severityBoost = severity === 'high' ? 0.1 : 0;
  return Math.min((matchRatio * 0.9 + severityBoost) * 100, 100);
}
```
